/**
 * Log service
 * Batches logs and stores in R2 (one object per 100 logs)
 */
 
/**
 * Store log in R2
 * @param {Object} logData - Log data
 * @param {R2Bucket} r2Bucket - R2 bucket
 * @param {KVNamespace} logState - KV namespace for tracking batch state
 * @returns {Promise<void>}
 */
export async function storeLog(logData, r2Bucket, logState) {
  try {
    console.log('[logService] storeLog called with logData:', JSON.stringify(logData));
    console.log('[logService] logState binding:', logState ? 'exists' : 'null');
    console.log('[logService] r2Bucket binding:', r2Bucket ? 'exists' : 'null');
    
    // Get current batch state
    const batchKey = 'current_batch';
    let batchState;
    
    try {
      console.log('[logService] Getting batch state from KV with key:', batchKey);
      const stateData = await logState.get(batchKey, { type: 'json' });
      console.log('[logService] KV get result:', stateData ? 'found existing batch' : 'no existing batch');
      batchState = stateData || {
        batchNumber: 1,
        logCount: 0,
        logs: [],
        lastLogTime: null,
      };
    } catch (error) {
      console.error('[logService] Error getting batch state from KV:', error.message, error.stack);
      // Initialize new batch state if KV read fails
      batchState = {
        batchNumber: 1,
        logCount: 0,
        logs: [],
        lastLogTime: null,
      };
    }
    
    console.log('[logService] Current batch state:', {
      batchNumber: batchState.batchNumber,
      logCount: batchState.logCount,
      logsLength: batchState.logs?.length || 0,
      lastLogTime: batchState.lastLogTime,
    });
    
    // Add log to batch
    batchState.logs.push({
      ...logData,
      timestamp: logData.timestamp || new Date().toISOString(),
    });
    batchState.logCount++;
    const now = Date.now();
    const lastLogTime = batchState.lastLogTime ? new Date(batchState.lastLogTime).getTime() : now;
    
    console.log('[logService] After adding log:', {
      logCount: batchState.logCount,
      logsLength: batchState.logs.length,
      now,
      lastLogTime,
      timeSinceLastLog: batchState.lastLogTime ? (now - lastLogTime) : 0,
    });
    
    // If batch is full (100 logs) OR batch has been sitting for more than 24 hours (86400000ms), save to R2
    // Changed from 5 minutes to 24 hours as per requirement
    const flushIntervalMs = 86400000; // 24 hours in milliseconds
    const shouldFlush = batchState.logCount >= 100 || 
      (batchState.lastLogTime && (now - lastLogTime) > flushIntervalMs);
    
    console.log('[logService] Should flush?', shouldFlush, {
      logCount: batchState.logCount,
      isFull: batchState.logCount >= 100,
      timeElapsed: batchState.lastLogTime ? (now - lastLogTime) : 0,
      timeThreshold: 86400000, // 24 hours
      timeThresholdHours: 24,
    });
    
    if (shouldFlush && batchState.logCount > 0) {
      const objectKey = `logs/batch-${batchState.batchNumber}-${now}.json`;
      
      try {
        await r2Bucket.put(objectKey, JSON.stringify(batchState.logs), {
          httpMetadata: {
            contentType: 'application/json',
          },
        });
        
        console.log(`[logService] Flushed ${batchState.logCount} logs to R2: ${objectKey}`);
        
        // Reset batch after successful write
        batchState.batchNumber++;
        batchState.logCount = 0;
        batchState.logs = [];
        batchState.lastLogTime = null;
      } catch (error) {
        console.error('[logService] Error writing to R2:', error.message, error.stack);
        // Don't reset batch if write failed - will retry next time
        // But still update lastLogTime so we don't keep trying immediately
        batchState.lastLogTime = new Date().toISOString();
      }
    } else {
      // Update last log time for time-based flushing
      batchState.lastLogTime = new Date().toISOString();
      console.log('[logService] Not flushing, updated lastLogTime:', batchState.lastLogTime);
    }
    
    // Update batch state in KV
    console.log('[logService] Saving batch state to KV...', {
      batchNumber: batchState.batchNumber,
      logCount: batchState.logCount,
      logsLength: batchState.logs?.length || 0,
    });
    
    try {
      const batchStateJson = JSON.stringify(batchState);
      console.log('[logService] Batch state JSON length:', batchStateJson.length);
      console.log('[logService] Batch state JSON (first 200 chars):', batchStateJson.substring(0, 200));
      
      const putResult = await logState.put(batchKey, batchStateJson);
      console.log('[logService] KV put result:', putResult);
      console.log('[logService] Successfully saved batch state to KV');
    } catch (error) {
      console.error('[logService] Error updating batch state in KV:', error.message, error.stack);
      // Re-throw to surface the error - we want to know if KV writes fail
      throw error;
    }
  } catch (error) {
    console.error('[logService] Error in storeLog:', error.message, error.stack);
    // Re-throw to surface the error
    throw error;
  }
}
 
/**
 * Flush remaining logs to R2
 * @param {R2Bucket} r2Bucket - R2 bucket
 * @param {KVNamespace} logState - KV namespace
 * @returns {Promise<void>}
 */
export async function flushLogs(r2Bucket, logState) {
  try {
    const batchKey = 'current_batch';
    let batchState;
    
    try {
      batchState = await logState.get(batchKey, { type: 'json' });
    } catch (error) {
      console.error('[logService] Error getting batch state from KV:', error.message);
      return;
    }
    
    if (batchState && batchState.logCount > 0) {
      const objectKey = `logs/batch-${batchState.batchNumber}-${Date.now()}.json`;
      
      try {
        await r2Bucket.put(objectKey, JSON.stringify(batchState.logs), {
          httpMetadata: {
            contentType: 'application/json',
          },
        });
        
        console.log(`[logService] Flushed ${batchState.logCount} logs to R2: ${objectKey}`);
        
        // Reset batch
        await logState.put(batchKey, JSON.stringify({
          batchNumber: batchState.batchNumber + 1,
          logCount: 0,
          logs: [],
          lastLogTime: null,
        }));
      } catch (error) {
        console.error('[logService] Error flushing logs to R2:', error.message, error.stack);
        throw error;
      }
    } else {
      console.log('[logService] No logs to flush');
    }
  } catch (error) {
    console.error('[logService] Error in flushLogs:', error.message, error.stack);
    throw error;
  }
}
